使用迭代器在容器和算法之间交互，程序清单16.1
```
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    vector <int> vecIntegerArray;
    vecIntegerArray.push_back(50);
    vecIntegerArray.push_back(2991);
    vecIntegerArray.push_back(23);
    vecIntegerArray.push_back(9999);

    cout << "The contents of the vector are: " << endl;

    vector <int>::iterator iArrayWalker = vecIntegerArray.begin();//iterator就是迭代器，有的地方叫它泛型指针

    while (iArrayWalker != vecIntegerArray.end())
    {
        cout << *iArrayWalker << endl;
        ++ iArrayWalker;
    }

    vector <int>::iterator iElement = find (vecIntegerArray.begin(), vecIntegerArray.end(), 23);//find是找23这个元素的指针
    if(iElement != vecIntegerArray.end())
    {
        int nPosition = distance(vecIntegerArray.begin(), iElement);//distance是计算两个指针的距离
        cout << "Value " << *iElement;
        cout << " found in the vector at position:" << nPosition << endl;
    }
    return 0;
}

```
